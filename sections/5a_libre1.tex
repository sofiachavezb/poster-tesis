\begin{itemize}

	\item \textbf{Factory:} Componentes desacoplados
	\item \textbf{Composite:} Pipelines jerárquicos
\end{itemize}

\separador

\textcolor{mikuDarkTurquoise}{\textbf{Validación:}}
\begin{itemize}
	\item \textbf{Completitud:} 6 dimensiones configurables
	\item \textbf{Extensibilidad:} Nuevos componentes sin modificar core
	\item \textbf{Reproducibilidad:} Pipelines de literatura instanciables
\end{itemize}
